    71→    71→function ScrollReveal({ children, className = '' }: { children: React.ReactNode; className?: string }) {
    72→    72→  const ref = useRef<HTMLDivElement>(null);
    73→    73→  const [isVisible, setIsVisible] = useState(false);
    74→    74→
    75→    75→  useEffect(() => {
    76→    76→    const el = ref.current;
    77→    77→    if (!el) return;
    78→    78→    const observer = new IntersectionObserver(
    79→    79→      ([entry]) => {
    80→    80→        if (entry.isIntersecting) {
    81→    81→          setIsVisible(true);
    82→    82→          observer.unobserve(el);
    83→    83→        }
    84→    84→      },
    85→    85→      { threshold: 0.08, rootMargin: '0px 0px -40px 0px' }
    86→    86→    );
    87→    87→    observer.observe(el);
    88→    88→    return () => observer.disconnect();
    89→    89→  }, []);
    90→    90→
    91→    91→  return (
    92→    92→    <div
    93→    93→      ref={ref}
    94→    94→      className={`transition-all duration-700 ease-out ${
    95→    95→        isVisible ? 'opacity-100 translate-y-0' : 'opacity-0 translate-y-8'
    96→    96→      } ${className}`}
    97→    97→    >
    98→    98→      {children}
    99→    99→    </div>
   100→   100→  );
   101→   101→}

   103→   103→function CounterAnimation({ target, duration = 2000 }: { target: number; duration?: number }) {
   104→   104→  const [count, setCount] = useState(0);
   105→   105→  const ref = useRef<HTMLSpanElement>(null);
   106→   106→  const hasAnimated = useRef(false);
   107→   107→
   108→   108→  useEffect(() => {
   109→   109→    const el = ref.current;
   110→   110→    if (!el) return;
   111→   111→    const observer = new IntersectionObserver(
   112→   112→      ([entry]) => {
   113→   113→        if (entry.isIntersecting && !hasAnimated.current) {
   114→   114→          hasAnimated.current = true;
   115→   115→          const startTime = performance.now();
   116→   116→          const animate = (currentTime: number) => {
   117→   117→            const elapsed = currentTime - startTime;
   118→   118→            const progress = Math.min(elapsed / duration, 1);
   119→   119→            // Ease-out cubic
   120→   120→            const eased = 1 - Math.pow(1 - progress, 3);
   121→   121→            setCount(Math.floor(eased * target));
   122→   122→            if (progress < 1) {
   123→   123→              requestAnimationFrame(animate);
   124→   124→            } else {
   125→   125→              setCount(target);
   126→   126→            }
   127→   127→          };
   128→   128→          requestAnimationFrame(animate);
   129→   129→          observer.unobserve(el);
   130→   130→        }
   131→   131→      },
   132→   132→      { threshold: 0.5 }
   133→   133→    );
   134→   134→    observer.observe(el);
   135→   135→    return () => observer.disconnect();
   136→   136→  }, [target, duration]);
   137→   137→
   138→   138→  return <span ref={ref}>{count}</span>;
   139→   139→}
